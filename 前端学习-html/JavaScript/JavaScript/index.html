使用示例

基本用法

// 访问对象属性
const name = person.name;
console.log(name); // 输出: John Doe

// 调用对象方法
person.sayHello(); // 输出: Hello, I'm John Doe

动态创建对象

// 使用构造函数创建对象
function createPerson(name, age) {
    return {
        name,
        age,
        sayHello() {
            console.log(`Hello, I'm ${this.name}`);
        }
    };
}

const person = createPerson("John Doe", 30);
